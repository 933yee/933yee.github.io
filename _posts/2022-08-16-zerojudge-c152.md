## [Question Link](https://zerojudge.tw/ShowProblem?problemid=c152 "ZeroJudge-c152")

**作法**
===
兩次 Dijkstra

**Code**
===

```cpp
#include <bits/stdc++.h>
using namespace std;
struct state{
    state(){}
    state(int i, float c): id(i), cost(c){}
    state(int i, float c, float d): id(i), cost(c), dis(d){}
    int id;
    float cost, dis;
    bool operator<(const state& n) const{
        return cost >= n.cost;
    }
};
struct node{
    node(){}
    node(int i, int d, float t):id(i), dis(d), time(t){}
    int id, dis;
    float time;
};
int main(){
    ios_base::sync_with_stdio(0);
    cin.tie(0);
    int t; cin >> t;
    while(t--){
        int N, M, A, B;
        cin >> N >> M >> A >> B;
        vector<vector<node>> neighbors(N+1);
        vector<bool> seen(N+1);
        while(M--){
            int a, b, d, v;
            cin >> a >> b >> d >> v; 
            neighbors[a].emplace_back(node(b, d, d/(float)v));
            neighbors[b].emplace_back(node(a, d, d/(float)v));
        }
        priority_queue<state> pq;
        // first dijkstra
        pq.emplace(state(A, 0));
        while(!pq.empty()){
            state cur = pq.top();
            pq.pop();
            if(seen[cur.id]) continue;
            if(cur.id == B){
                cout << (int)cur.cost << ' ';
                break;
            }
            seen[cur.id] = true;
            for(node& i:neighbors[cur.id]){
                if(!seen[i.id]){
                    pq.emplace(state(i.id, cur.cost + i.dis));
                }
            }
        }
        // init 
        while(!pq.empty()) pq.pop();
        for(int i=0; i<=N; i++) seen[i] = false;
        // second dijkstra
        pq.emplace(state(A, 0, 0));
        while(!pq.empty()){
            state cur = pq.top();
            pq.pop();
            if(seen[cur.id]) continue;
            if(cur.id == B){
                cout << (int)cur.dis << '\n';
                break;
            }
            seen[cur.id] = true;
            for(node& i:neighbors[cur.id]){
                if(!seen[i.id]){
                    pq.emplace(state(i.id, cur.cost + i.time, cur.dis + i.dis));
                }
            }
        }
    }
}
```
