**作法**
===
Dijkstra + DP

**Code**

```cpp
#include <bits/stdc++.h>
using namespace std;
struct node{
    int id, cost;
    node(){}
    node(int i, int c):id(i),cost(c){}
    bool operator<(const node& a)const{
        return cost > a.cost;
    }
};
vector<vector<int>> neighbors;
vector<bool> seen;
vector<int> shortest, dp;
int path[1050][1050];
void init(int N){
    dp.clear();
    dp.resize(N+1, -1);
    neighbors.clear();
    neighbors.resize(N+1);
    seen.clear();
    seen.resize(N+1);
    shortest.clear();
    shortest.resize(N+1, INT_MAX);
    shortest[2] = 0;
}
int dfs(int id){
    if(id == 2) return 1;
    if(~dp[id]) return dp[id];
    int s = 0;
    for(int& i:neighbors[id]){
        if(shortest[i] < shortest[id])
            s += dfs(i);
    }
    dp[id] = s;
    return s;
}
int main(){
    ios_base::sync_with_stdio(0);
    cin.tie(0);
    int N, M;
    while(cin >> N, N){
        cin >> M;
        init(N);
        while(M--){
            int a, b, d;
            cin >> a >> b >> d;
            neighbors[a].emplace_back(b);
            neighbors[b].emplace_back(a);
            path[a][b] = path[b][a] = d;
        }
        priority_queue<node> pq;
        pq.emplace(node(2, 0));
        while(!pq.empty()){
            node cur = pq.top();
            pq.pop();
            if(seen[cur.id]) continue;
            else seen[cur.id] = true;
            for(int& i:neighbors[cur.id]){
                if(!seen[i]){
                    if(cur.cost + path[i][cur.id] < shortest[i]){
                        shortest[i] = cur.cost + path[i][cur.id];
                        pq.emplace(node(i, shortest[i]));
                    }
                }
            }
        }
        cout << dfs(1) << '\n';
    }
}
```
